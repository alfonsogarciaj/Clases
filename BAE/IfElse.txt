create procedure pa_mayorqueP( in provincia1 varchar(20), in provincia2 varchar(20)) 
begin declare cant1 int; declare cant2 int; 
select count(*) into cant1 from clientes as C join provincias as p on p.codigo=C.codigoprovincia where p.nombre=provincia1; 
select count(*) into cant2 from clientes as C join provincias as p on p.codigo=C.codigoprovincia where p.nombre=provincia2; 
if cant1>cant2 then 
select provincia1, cant1; 
elseif cant2>cant1 then 
select provincia2,cant2; 
else select provincia1, cant1, provincia2, cant2; 
end if;
end //
